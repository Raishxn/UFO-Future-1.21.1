    AE2 GenericStack

    { "#": 15000000, "#t": "ae2:i", "id": "minecraft:iron_ingot" }
    • # = shu liang
    • #t = lei xing: ae2:i biao shi wu pin, ae2:f biao shi liu ti
    • id = wan zheng registry id
